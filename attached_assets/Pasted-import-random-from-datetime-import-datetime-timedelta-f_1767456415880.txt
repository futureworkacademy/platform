import random
from datetime import datetime, timedelta
from polygon import RESTClient  # For macro data proxy

# Function to get macro modifier (as before)
def get_macro_modifier():
    client = RESTClient()  # Assumes API key in env
    today = datetime.now().strftime("%Y-%m-%d")  # Use real today for proxy, even if sim in 2026
    try:
        sp500 = client.get_aggs("SPY", 1, "day", today, today)[0]
        percent_change = ((sp500.close - sp500.open) / sp500.open) * 100
        if percent_change > 0.5:
            return 1.10
        elif percent_change < -0.5:
            return 0.90
        else:
            return 1.00
    except:
        return 1.00

# Initial company state (added lobbying budget)
company = {
    'revenue': 1000000,
    'employees': 500,
    'morale': 80,
    'ai_budget': 200000,
    'reskilling_fund': 100000,
    'lobbying_budget': 50000  # New: For influencing events/regulations
}

departments = [
    {'name': 'Operations', 'ai_option': 'Automate inventory with ML', 'job_impact': -50, 'revenue_boost': 200000, 'risk': 20},
    {'name': 'Sales', 'ai_option': 'Predictive analytics for leads', 'job_impact': -20, 'revenue_boost': 150000, 'risk': 15},
    {'name': 'Customer Service', 'ai_option': 'AI chatbots for support', 'job_impact': -30, 'revenue_boost': 100000, 'risk': 25},
    {'name': 'R&D', 'ai_option': 'ML for product innovation', 'job_impact': 20, 'revenue_boost': 300000, 'risk': 30}
]

macro_modifier = get_macro_modifier()

def apply_event(event, lobbied=False):
    print(f"\n*** Global Event: {event['name']} ***\n{event['desc']}")
    impact = event['impact'].copy()
    if lobbied:
        # Lobbying mitigates negative impacts by 50%
        for key in impact:
            if impact[key] < 0:
                impact[key] *= 0.5
        print("Lobbying efforts softened the blow!")
    if 'revenue' in impact:
        company['revenue'] *= (1 + impact['revenue'])
    if 'morale' in impact:
        company['morale'] += impact['morale']
    if 'employees' in impact:
        company['employees'] += impact['employees']
    if 'job_impact' in impact:
        company['employees'] += impact['job_impact']
    company['morale'] = max(0, min(100, company['morale']))

def generate_briefing():
    num_articles = random.randint(2, 4)
    briefing = "\n--- Weekly Briefing Package ---\nRead and digest these materials for insights:\n"
    selected = random.sample(briefing_templates, num_articles)
    for article in selected:
        briefing += f"Title: {article['title']}\nContent: {article['content']}\nPossible Insights: {', '.join(article['insights'])}\n\n"
    briefing += "Spend time analyzing these for better decisions!\n"
    return briefing

def simulate_week(week_num, current_date):
    print(f"\n--- Week {week_num} ({current_date.strftime('%Y-%m-%d')}) ---")
    print(generate_briefing())  # 30-min activity: Read and think
    print(f"Macro Modifier: x{macro_modifier:.2f}")
    print(f"Current Stats: Revenue ${company['revenue']:,} | Employees: {company['employees']} | Morale: {company['morale']}% | AI Budget: ${company['ai_budget']:,} | Reskilling: ${company['reskilling_fund']:,} | Lobbying: ${company['lobbying_budget']:,}")
    
    # Random global event (30% chance now for more dynamism)
    if random.random() < 0.30:
        event = random.choice(global_events)
        if input("Spend lobbying budget to mitigate? (y/n, cost $10,000): ").lower() == 'y' and company['lobbying_budget'] >= 10000:
            company['lobbying_budget'] -= 10000
            apply_event(event, lobbied=True)
        else:
            apply_event(event)
    
    # Ad hoc event trigger (optional)
    if input("\nTrigger ad hoc event? (y/n): ").lower() == 'y':
        print("Choose event:")
        for i, ev in enumerate(global_events, 1):
            print(f"{i}. {ev['name']}")
        choice = int(input("Enter number: ")) - 1
        apply_event(global_events[choice])
    
    # Lobbying option: Spend to boost positives or prevent negatives
    if input("\nSpend on general lobbying (e.g., for AI incentives, cost $5,000)? (y/n): ").lower() == 'y' and company['lobbying_budget'] >= 5000:
        company['lobbying_budget'] -= 5000
        boost = random.choice([0.05, 0.10])  # Revenue or morale boost
        if random.random() < 0.5:
            company['revenue'] *= (1 + boost)
            print(f"Lobbying success: Revenue boosted by {boost*100}%!")
        else:
            company['morale'] += int(boost * 100)
            print(f"Lobbying success: Morale boosted by {int(boost*100)}%!")

    # AI Deployment
    print("\nChoose department for AI/ML (0 to skip):")
    for i, dept in enumerate(departments, 1):
        print(f"{i}. {dept['name']}: {dept['ai_option']}")
    choice = int(input("Enter number: ")) - 1
    if choice == -1:
        return
    dept = departments[choice]
    cost = random.randint(50000, 100000)
    if cost > company['ai_budget']:
        print("Insufficient AI budget!")
        return
    company['ai_budget'] -= cost
    boosted_revenue = dept['revenue_boost'] * macro_modifier
    boosted_job = int(dept['job_impact'] * macro_modifier)
    company['revenue'] += boosted_revenue
    company['employees'] += boosted_job
    
    # Transitions
    if boosted_job < 0:
        transitions = abs(boosted_job)
        reskill_cost = transitions * 2000
        print(f"Job losses: {transitions}. Reskill cost: ${reskill_cost:,}")
        if input("Fund reskilling? (y/n): ").lower() == 'y':
            if reskill_cost <= company['reskilling_fund']:
                company['reskilling_fund'] -= reskill_cost
                morale_impact = -5
            else:
                morale_impact = -20
        else:
            morale_impact = -30
    else:
        morale_impact = 10
    
    # Risk
    if random.randint(1, 100) <= dept['risk']:
        print("Risk: AI issue!")
        company['revenue'] -= 50000
        morale_impact -= 15
    
    company['morale'] += morale_impact
    company['morale'] = max(0, min(100, company['morale']))

# Main simulation

    current_date = start_date + timedelta(weeks=week-1)
    simulate_week(week, current_date)
